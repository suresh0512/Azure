  name: virtualNetworkName
  location: location
  tags: {}
  properties: {
    addressSpace: {
      addressPrefixes: [
        addressSpaces
      ]
    }
    subnets: [
      {
        name: subnet0_name
        properties: {
          addressPrefix: subnet0_addressRange
        }
      }
    ]
  }
  dependsOn: []
}
